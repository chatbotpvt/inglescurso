    const chatBox = document.getElementById('chat-box');
    const input = document.getElementById('user-input');

    async function sendMessage() {
      const userMessage = input.value.trim();
      if (!userMessage) return;

      chatBox.innerHTML += `<div><span class="user">T√∫:</span> ${userMessage}</div>`;
      input.value = '';

      const response = await fetch("https://api.openai.com/v1/chat/completions", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          "Authorization": "Bearer TU_API_KEY" // ‚Üê REEMPLAZA AQU√ç
        },
        body: JSON.stringify({
          model: "gpt-3.5-turbo",
          messages: [
            { role: "system", content: "Eres un profesor de ingl√©s amigable que ense√±a con ejemplos, frases √∫tiles y conversaci√≥n b√°sica." },
            { role: "user", content: userMessage }
          ]
        })
      });

      const data = await response.json();
      const botReply = data.choices?.[0]?.message?.content || "Lo siento, hubo un error.";

      chatBox.innerHTML += `<div><span class="bot">Chatbot:</span> ${botReply}</div>`;
      chatBox.scrollTop = chatBox.scrollHeight;
    }
  </script>
